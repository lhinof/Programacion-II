Ejemplo de Programación Orientada a Objetos

class Animal {
  constructor(nombre, tipo) {
    this.nombre = nombre;
    this.tipo = tipo;
  }

  hablar() {
    return `Soy un ${this.tipo} y mi nombre es ${this.nombre}.`;
  }
}

class Perro extends Animal {
  constructor(nombre) {
    super(nombre, 'perro');
  }

  hablar() {
    return `${this.nombre} dice ¡Guau!`;
  }
}

class Gato extends Animal {
  constructor(nombre) {
    super(nombre, 'gato');
  }

  hablar() {
    return `${this.nombre} dice ¡Miau!`;
  }
}

const perro1 = new Perro('Max');
const gato1 = new Gato('Git');

console.log(perro1.hablar());
console.log(gato1.hablar());
